An example of a main.c that starts everything up



#include "my_stuff.h"

#include <robusto_logging.h>
#include <robusto_init.h>
#include <robusto_network_init.h>
#include <robusto_server_init.h>
#include <robusto_misc_init.h>

char * log_prefix = "MyStuff";

void setup()
{
    ROB_LOGI(log_prefix, "Register services.");
    register_network_service();
    register_server_service();
    ROB_LOGI(log_prefix, "Starting Robusto.");
    init_robusto();
    ROB_LOGI(log_prefix, "Initializing my stuff.");
    init_my_stuff();
}


void app_main() {
   setup();
   ROB_LOGI(log_prefix, "Doin' my stuff.");
   do_my_stuff();
}
